import { TestBed } from '@automock/jest';

class Service1 {}
class Service2 {}
class Dependency1 {}

describe('multiple TestBed patterns', () => {
  it('should support different styles', () => {
    // Style 1: One-liner
    const test1 = TestBed.create(Service1).compile();
    
    // Style 2: With mock override
    const test2 = TestBed.create(Service2)
      .mock(Dependency1)
      .using({ method: jest.fn() })
      .compile();

    // Style 3: Variable assignment first
    const builder = TestBed.create(Service1);
    const test3 = builder.compile();
      
    expect(test1.unit).toBeDefined();
    expect(test2.unit).toBeDefined();
    expect(test3.unit).toBeDefined();
  });
});
