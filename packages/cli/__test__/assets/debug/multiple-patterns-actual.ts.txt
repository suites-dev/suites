import { TestBed } from "@suites/unit";
class Service1 {}
class Service2 {}
class Dependency1 {}
describe('multiple TestBed patterns', () => {
  it('should support different styles', async () => {
    // Style 1: One-liner
    const test1 = await TestBed.solitary(Service1).compile();

    // Style 2: With mock override
    const test2 = await TestBed.solitary(Service2).mock(Dependency1).impl(stubFn => ({
      method: stubFn()
    })).compile();

    // Style 3: Variable assignment first
    const builder = await TestBed.solitary(Service1);
    const test3 = await builder.compile();
    expect(test1.unit).toBeDefined();
    expect(test2.unit).toBeDefined();
    expect(test3.unit).toBeDefined();
  });
});
